2.Print a given number, in reverse order using a Shell script such that the input is provided using command Line Argument only.


#!/bin/sh


echo "enter the number"
read n
s=0
while [ $n -gt 0 ]
do
    r = `expr $n % 10`
    s = `expr ($s \* 10) + $r`
    n = `expr $n / 10`
done
echo "reverse no is $s”
